\documentclass[11pt,a4paper]{article}

% ---------------------------
% Unicode font (Times New Roman)
% ---------------------------
\usepackage{fontspec}
\setmainfont{TeX Gyre Termes}  % main text

% ---------------------------
% Page layout and spacing
% ---------------------------
\usepackage{geometry}
\geometry{top=1cm, bottom=1cm, left=1cm, right=1cm}
\usepackage{parskip}           % add space between paragraphs

% ---------------------------
% Headings and lists
% ---------------------------
\usepackage{titlesec}
\usepackage{enumitem}

\titleformat{\section}{\Huge\bfseries}{}{0em}{}
\titleformat{\subsection}{\LARGE\bfseries}{}{0em}{}
\titleformat{\subsubsection}{\Large\bfseries}{}{0em}{}
\titleformat{\paragraph}{\large\bfseries}{}{0em}{}
\titleformat{\subparagraph}{\normalsize\bfseries}{}{0em}{}

\setlist[itemize]{noitemsep, topsep=0pt}

% ---------------------------
% Hyperlinks with real underline
% ---------------------------
\usepackage[normalem]{ulem} % allows underlining without affecting emphasis
\usepackage{xcolor}         % for colors if needed
\usepackage{hyperref}

\hypersetup{
    colorlinks=true,  % make links clickable
    linkcolor=black,  % black link text
    urlcolor=black,
    citecolor=black
}

% Redefine \href to automatically underline text
\let\oldhref\href
\renewcommand{\href}[2]{\uline{\oldhref{#1}{#2}}}


% ---------------------------
% Remove headers and footers
% ---------------------------
\pagestyle{empty}

% ---------------------------
% Tightlist fix for Pandoc
% ---------------------------
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% ---------------------------
% Prevent page breaks in blocks
% ---------------------------
\usepackage{needspace}

% ---------------------------
% Title (Name)
% ---------------------------
$if(title)$
\makeatletter
\def\maketitle{
  \begin{center}
    {\LARGE \bfseries \@title \par}
  \end{center}
  \vspace{1em}
}
\makeatother
\title{$title$}
\maketitle
$endif$

% ---------------------------
% Begin document
% ---------------------------
\begin{document}

$include-before$
$endfor$

$body$

$for(include-after)$
$include-after$
$endfor$
